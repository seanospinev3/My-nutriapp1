exports.handler = async (event) => {
  if (event.httpMethod === "OPTIONS") {
    return {
      statusCode: 200,
      headers: {
        "Access-Control-Allow-Origin": "*",
        "Access-Control-Allow-Headers": "Content-Type",
        "Access-Control-Allow-Methods": "POST, OPTIONS",
      },
      body: "",
    };
  }

  if (event.httpMethod !== "POST") {
    return { statusCode: 405, body: "Method Not Allowed" };
  }

  try {
    const body = JSON.parse(event.body);

    // Convert Anthropic format → Gemini format
    const userMessage = body.messages?.[0]?.content;
    const textContent = Array.isArray(userMessage)
      ? userMessage.filter(c => c.type === "text").map(c => c.text).join("\n")
      : userMessage || "";

    // Handle image if present (for Vision scanner)
    const imagePart = Array.isArray(userMessage)
      ? userMessage.find(c => c.type === "image")
      : null;

    const parts = [];
    if (imagePart) {
      parts.push({
        inline_data: {
          mime_type: imagePart.source.media_type,
          data: imagePart.source.data,
        },
      });
    }
    parts.push({ text: textContent });

    const geminiRes = await fetch(
      `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${process.env.GEMINI_API_KEY}`,
      {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ contents: [{ parts }] }),
      }
    );

    const geminiData = await geminiRes.json();
    const text = geminiData.candidates?.[0]?.content?.parts?.[0]?.text || "No response";

    // Return in Anthropic-compatible format so the app doesn't need changes
    return {
      statusCode: 200,
      headers: {
        "Content-Type": "application/json",
        "Access-Control-Allow-Origin": "*",
      },
      body: JSON.stringify({
        content: [{ type: "text", text }],
      }),
    };
  } catch (err) {
    return {
      statusCode: 500,
      headers: { "Access-Control-Allow-Origin": "*" },
      body: JSON.stringify({ error: err.message }),
    };
  }
};
